```js 数组中找出两数之和等于目标值m
function twoSumM(nums, m) {
  let map = new Map();
  for (let num of nums) {
    let temp = m - num;
    if (map.has(temp)) return [temp, num];
    else map.set(num, num);
  }
  return null;
}
```
